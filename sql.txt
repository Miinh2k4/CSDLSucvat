CREATE TABLE bang
(
 id int primary key,
 name varchar(30),
 age int,
 address varchar(50),
 gpa int,
 ComboID int,
FOREIGN KEY (ComboID) REFERENCES category(ComboID)
)


CREATE TABLE bangid(
    ComboID INT PRIMARY KEY
);
